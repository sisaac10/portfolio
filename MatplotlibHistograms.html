import matplotlib.pyplot as plt
from pandas import Series, DataFrame
import pandas as pd


def main():
    print()
    print("This is a program that will produce histograms for height and weight")
    print("data taken from http://wiki.stat.ucla.edu/socr/index.php/SOCR_Data_Dinov_020108_HeightsWeights")
    print("The data was copied and stored in Excel file assign11python.xlsx")
    print()
#------------------------------------------------------------------------------
    t = {}
    xls_file = pd.ExcelFile('assign11python.xlsx')     
    t = xls_file.parse("Sheet1")
#------------------------------------------------------------------------------    
                                        #create empty series for heights and weights
    height = Series([])
    weight = Series([])
    
    for i in range(len(t)):             #create for loop to populate empty series with heights and weights
        height[i] = t.ix[i][1]
        weight[i] = t.ix[i][2]
    
#------------------------------------------------------------------------------       
    fig1 = plt.figure()                               #create separate plots for heights and weights
    hw1 = fig1.add_subplot(1,1,1)
    hw1.hist(height, bins = 16) 
    hw1.set_title('Histogram of Heights')
    hw1.set_ylabel('Number of Individuals')
    hw1.set_xlabel('Height (Inches)')
#-----------------------------------------------------------------------------    
    fig2 = plt.figure()
    hw2 = fig2.add_subplot(1,1,1)
    hw2.hist(weight, bins = 16, color = "red")
    hw2.set_title('Histogram of Weights')
    hw2.set_ylabel('Number of Individuals')
    hw2.set_xlabel('Weight (Inches)')
#------------------------------------------------------------------------------
    plt.show()                                       #show first and second plot
    
main()
