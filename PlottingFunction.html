Plotting a Function¶

This is a program that uses matplotlib to plot the function f(x) = e^(-x/p)*sin(pi/x) over the interval [0, 10], for the values of the parameter p = 1, 2, 5. Labels are included for x and y axes, as well as a legend to mark corresponding lines with their plots.

import matplotlib.pyplot as plt
from pandas import Series, DataFrame
import pandas as pd
import numpy as np

def main():
    print()
    print("This is a program that plots the function f(x) = e**(-x/p)sin(pi/x)")
    print("over the interval [0,10] for the values of the parameter p = 1,2,5")
    print()
#-----------------------------------------------------------------------------
    p = 1                                                     #create curve for p = 1
    x = np.linspace(0,10,100)
    curve1 = np.exp(-x/p)*np.sin(np.pi*x)
        
    fig1 = plt.figure()
    
    hw1 = fig1.add_subplot(1,1,1)
    hw1.plot(x,curve1, label = 'p = 1')
#-----------------------------------------------------------------------------    
    p = 2                                                   #create curve for p = 2
    curve2 = np.exp(-x/p)*np.sin(np.pi*x)
    
    hw2 = fig1.add_subplot(1,1,1)
    hw2.plot(x,curve2, label = 'p = 2')
#-----------------------------------------------------------------------------    
    p = 5                                                   #create curve for p = 5
    curve3 = np.exp(-x/p)*np.sin(np.pi*x)
    
    hw3 = fig1.add_subplot(1,1,1)
    hw3.plot(x,curve3, label = 'p = 5')
#-----------------------------------------------------------------------------    
    hw1.set_title('Plotting a Function')                   #create labels and legend
    hw1.legend(loc = 'best')
    hw1.set_ylabel('f(x)')
    hw1.set_xlabel('interval [0,10]')
#-----------------------------------------------------------------------------    
    plt.show()                                             #show plot
    
    
main()
