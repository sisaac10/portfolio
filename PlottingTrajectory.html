import warnings
import matplotlib.pyplot as plt
from pandas import Series, DataFrame
import pandas as pd
import numpy as np

def main():
    print()
    print("This is a program that plots the trajectories of a projectile")
    print()
    warnings.filterwarnings("ignore")
#-----------------------------------------------------------------------------    
    V = 100                                                #set values for variables time, velocity, gravity, and theta
    t = np.linspace(0, 25, 100)
    g = 9.8
    
    theta = np.arange(0, np.pi/2, np.pi/16 )
#-----------------------------------------------------------------------------    
    fig1 = plt.figure()                                    #create figure and set x y limits for plot
    
    hw1 = fig1.add_subplot(1,1,1)
    hw1.set_ylim([-50,600])
    hw1.set_xlim([-50, 1200])
#-----------------------------------------------------------------------------    
    for i in theta[1:8]:                                     #for loop to create and populate array with xy coordinates for plot
        x1 = []
        y1 = []
        for k in t:
            x = V*k*np.cos(i)
            y = (V*k*np.sin(i)) - ((g*k**2)/2)
            if x >= 0 and y >=0:
                x1.append(x)
                y1.append(y)
#-----------------------------------------------------------------------------                
        hw1.plot(x1,y1, label = (i*(180/np.pi)))            #plot values and create lables and legend
        hw1.legend(loc = 'best')
        hw1.set_title('Plotting the trajectory of a projectile')
    
        hw1.set_xlabel('V*t*cos(theta)')
        hw1.set_ylabel('V*t*sin(theta) - (gt**2)/2')
#-----------------------------------------------------------------------------
        
        fig1.show()                                         #show plot
        
        
    
main()
