def insertion(list1):
    
    t5 = timer()
    for i in range(1,len(list1)):
        val = list1[i]
        j = i - 1
        while j >= 0:
            if val < list1[j]:
                list1[j + 1] = list1[j]
                list1[j] = val
                j = j - 1
            else:
                break
    t6 = timer()
    z = t6 - t5
    return z

def merge(arr1, arr2):
    i = j = 0
    newA = []
    l1 = len(arr1)
    l2 = len(arr2)
    while i < l1 or j < l2:
        if i < l1 and j < l2:
            if arr1[i] <= arr2[j]:
                newA.append(arr1[i])
                i += 1
            else:
                newA.append(arr2[j])
                j += 1
        elif i < l1:
            newA.append(arr1[i])
            i += 1
        else:
            newA.append(arr2[j])
            j += 1
    return newA

def partition(A, start, end):
    end2 = 0
    end2 = int(end - 1)
    pivot = end2
    divider = start
    
    for i in range(start, end2):
        if A[i] < A[pivot]:
            A[i], A[divider] = A[divider], A[i]
            divider += 1
            
    A[pivot], A[divider] = A[divider], A[pivot]
    
    return divider

def quicksort(A, start, end):
    
    t1 = timer()
    if (start < end):
        p = partition(A, start, end)
        quicksort(A, start, p - 1)
        quicksort(A, p + 1, end)
    t2 = timer()
    x = t2 - t1
